#include<stdio.h>
int main()
{
    int arr[100];
    int n;
    //int count=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
     for (int i=0;i<n;i++)

            }
        }

        }
     }

}
