#include<stdio.h>
int main()
{
    int n=5;
    int sum=0;
    int arr[n];
    for(int i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
        sum+=arr[i];
    }
     int min,max;
     min=max=arr[0];
     for (int i=0;i<n;i++)
     {
        if (arr[i]<min) min=arr[i];
        if (arr[i]>max) max=arr[i];
     }
    int min_sum=sum-max;
    int max_sum=sum-min;
    printf("%ld %ld",min_sum ,max_sum);
    return 0;
    
}
