/* Plus Minus ratio */
#include<stdio.h>
int main()
{
    int arr[100];
    int n;
    int countpos=0,countneg=0,countzero=0;
    //n=sizeof(arr)/sizeof(int);
    scanf("%d",&n);
    //printf("enter the array elements");
    for(int i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    for(int i=0;i<n;i++)
    {
        if (arr[i]>0)
        {
            countpos++;
        }
        else if (arr[i]<0)
        {
            countneg++;
        }
        else 
        {
            countzero++;
        }
    }
    float ratiopos=(float)countpos/n;
    float rationeg=(float)countneg/n;
    float ratiozero=(float)countzero/n;
    printf("%f\n%f\n%f\n",ratiopos,rationeg,ratiozero);
    return 0;
}
