#include<stdio.h>
int main()
{
    int n;
    scanf("%d",&n);
    int ar[n];
    int fre[1001]={0};
    for(int i=0;i<n;i++)
    {
        scanf("%d ",&ar[i]);
    }
    for (int i=0;i<n;i++)
    {
        fre[ar[i]]++;
    }
    int pair=0;
    for(int i=0;i<1000;i++)
    {
    pair=pair+ fre[i]/2;
  
    }
    printf("%d",pair);
    
}
